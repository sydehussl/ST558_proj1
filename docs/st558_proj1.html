Validating Numeric Variables

For numeric variables, we need to consider following:

-The user has to leave the default AGEP or specify their own selection of variables -The additional variables they specify must be valid -PWGTP should always be returned regardless of the user’s selection -There must be at least two numeric variables returned

To satisfy these conditions, we should:

  1. Make sure the user doesn’t pass NULL as their selection
  2. Make sure each variable specified is valid
  3. Add PWGTP to the vector of selected variables

Adding this logic to our function, we have:

query.census <- function(year = 2022, numeric_vars = "AGEP", categorical_vars = "SEX", 
                         geography = "State", specific_geo = "") {
  # check for valid year
  if(year == 2020 | !(year %in% 2010:2022)) stop("Error: Invalid year. Specify a year between 2010 and 2022 (not 2020)")
  
  # check for valid numeric vars
  numeric_vars_opts <- c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP")
  
  if(length(numeric_vars) == 0) stop("No numeric variable selected. Specify at least one.")
  
  for(i in numeric_vars) {
    if(!(i %in% numeric_vars_opts)) {
      stop("Invalid numeric variable. Specify a single variable or a vector of variables. Options: AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP")
    }
  }
  
  # make sure PWGTP ends up in final selection
  numeric_vars <- c(numeric_vars, "PWGTP")
  
  # return numeric vars for testing
  return(numeric_vars)
}

Validating Geography

We have to consider the following:

  • geography must be State, Region, or District
  • specific_geo must be between 1 and 50 for state, 1 and 4 for region, and 1 and 9 for division
  • specific_geo can be “all” for any selection of geography

Adding these considerations to our function:

  geography_opts <- c("Region", "State", "Division")
  if(!(geography %in% geography_opts)) stop("Invalid geography classification. Specify one of: State, Region, Division")

  }
  
  # build API URL
  api_key <- "59c3be738e0d051c63b57b6327de0b11ef0f7a1e"
  api_url <- paste0("https://api.census.gov/data/",
                    year,
                    "/acs/acs1/pums?get=",

fix.time <- function (time) {
  suppressPackageStartupMessages(library(lubridate))
  
  # don't mess up NA values

  # return vector converted to 24 hr time format

fix.time <- function (time, type) {
  library(lubridate)
  
  is_zero <- time == 0
  
  if(type == "JWAP") {
    scnds_since_midnight <- 120 + (time-1)*300 + ifelse(time > 6, 150, 0) + 
                            ifelse(time > 7, 150, 0) + ifelse(time > 9, 150, 0) +
                            ifelse(time > 10, 150, 0) + ifelse(time > 20, 150, 0) +
                            ifelse(time > 21, 150, 0)
  } else if(type == "JWDP") {
    scnds_since_midnight <- 900 + min(5, time-1)*1800 + 
           ifelse(time > 6, 1170 + min(11, time - 7)*600, 0) +
           ifelse(time > 18, 450 + min(59, time - 19)*300, 0) +
           ifelse(time > 78, 450 + min(53, time - 79)*600, 0) +
           ifelse(time > 132, 1200 + min(3, time - 133)*1800, 0) +
           ifelse(time > 136, 1200 + min(11, time - 137)*600, 0) +
           ifelse(time > 148, 1200 + min(2, time - 149)*1800, 0)
  } else {
    stop("Invalid type entered. Enter either 'JWAP' or 'JWDP'.")
  }
  
  scnds_since_midnight[is_zero] <- 0
  
  return(hms::as_hms(scnds_since_midnight))

    warning("Warning: 2020 included in year selection. 2020 Data is not inluded in the output.")
  }
  
  output <- data.frame()
  
  # never include 2020 regardless of user selection
  for(i in years[years != 2020]) {
    year_data <- query.census(year = i, numeric_vars = numeric_vars, categorical_vars = categorical_vars,
                              geography = geography, specific_geo = specific_geo)
    
    year_data$year <- i
    
    output <- rbind(output, year_data)
  }
  
  return(output)  
}

Time Fixing Helper Function

fix.time <- function (time, type) {
  library(lubridate)
  
  is_zero <- time == 0
  
  if(type == "JWAP") {
    scnds_since_midnight <- 120 + (time-1)*300 + ifelse(time > 6, 150, 0) + 
                            ifelse(time > 7, 150, 0) + ifelse(time > 9, 150, 0) +
                            ifelse(time > 10, 150, 0) + ifelse(time > 20, 150, 0) +
                            ifelse(time > 21, 150, 0)
  } else if(type == "JWDP") {
    scnds_since_midnight <- 900 + min(5, time-1)*1800 + 
           ifelse(time > 6, 1170 + min(11, time - 7)*600, 0) +
           ifelse(time > 18, 450 + min(59, time - 19)*300, 0) +
           ifelse(time > 78, 450 + min(53, time - 79)*600, 0) +
           ifelse(time > 132, 1200 + min(3, time - 133)*1800, 0) +
           ifelse(time > 136, 1200 + min(11, time - 137)*600, 0) +
           ifelse(time > 148, 1200 + min(2, time - 149)*1800, 0)
  } else {
    stop("Invalid type entered. Enter either 'JWAP' or 'JWDP'.")
  }
  
  scnds_since_midnight[is_zero] <- 0
  
  return(hms::as_hms(scnds_since_midnight))
}

Summary Function

summary.census <- function(census, numeric_vars = "all", categorical_vars = "all") {
  valid_numeric <- names(select(census, where(is.numeric)))
  valid_categorical <- names(select(census, where(is.factor)))
  invalid_vars <- c("PWGTP", "JWAP", "JWDP", "year")
  
  # validate numeric vars
  if(numeric_vars == "all") {
    numeric_vars = valid_numeric
  } else if(length(setdiff(numeric_vars, valid_numeric)) > 0) {
    stop("At least one numeric variable requested is not in the census data provided")
  }
  
  # remove PWGTP/year from numeric vars to summarizr
  numeric_vars <- setdiff(numeric_vars, invalid_vars)
    
  # validate categorical vars
  if(categorical_vars == "all") {
    categorical_vars = valid_categorical
  } else if(length(setdiff(categorical_vars, valid_categorical)) > 0) {
    stop("At least one categorical variable requested is not in the census data provided")
  }
  
  # compute numeric stats
  numeric_stats <- census |>
                   summarize(across(all_of(numeric_vars), list(
                          mean = ~ sum(.x*PWGTP)/sum(PWGTP),
                          sd = ~ sqrt(sum((.x-(sum(.x*PWGTP)/sum(PWGTP)))^2 * PWGTP) / sum(PWGTP))),
                          .names = "{.col}_{.fn}"))
  
  # doo doo horrible convoluted way of computing categorical stats (it works though)
  categorical_stats <- census |>
                       select(all_of(categorical_vars)) |>
                       pivot_longer(everything(), names_to = "var", values_to = "level") |>
                       count(var, level, name = "n") |>
                       mutate(colname = paste0(var, "_", level, "_n")) |>
                       select(colname, n) |>
                       pivot_wider(names_from = colname, values_from = n)
  
  # return results
  return(c(numeric_stats, categorical_stats))
}

Plotting Function

plot.census <- function(census, numeric_var, categorical_var) {
  suppressPackageStartupMessages(library(dplyr))
  library(ggplot2)
  
  valid_numeric <- names(select(census, where(is.numeric)))
  valid_categorical <- names(select(census, where(is.factor)))
  invalid_vars <- c("PWGTP", "JWAP", "JWDP")
  
  # validate numeric vars
  if(length(numeric_var) > 1) {
    stop("Multiple numeric variables specified. Specify only one.")
  } else if(!(numeric_var %in% valid_numeric)) {
    stop("The numeric variable requested is not in the census data provided")
  } else if(numeric_var %in% invalid_vars) {
    stop("The numeric variable requested is invalid. Invalid variables: JWAP, JWDP, PWGTP")
  }
    
  # validate categorical vars
  if(length(categorical_var) > 1) {
    stop("Multiple categorical variables specified. Specify only one.")
  } else if(!(categorical_var %in% valid_categorical)) {
    stop("At least one categorical variable requested is not in the census data provided")
  }
  
  # build title string
  title <- paste("Distribution of",
                 numeric_var,
                 "by",
                 categorical_var)
  
  # plot!
  plot <- ggplot(census, aes(y = get(numeric_var), 
                             fill = get(categorical_var), 
                             weight = PWGTP)) +
          geom_boxplot() +
          labs(title = title,
               x = "",
               y = numeric_var,
               fill = categorical_var) +
          scale_x_discrete(labels = NULL, breaks = NULL) +
          theme_minimal()
  
  return(plot)
}
