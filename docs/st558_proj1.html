query.census <- function(year = 2022, numeric_vars = "AGEP", categorical_vars = "SEX",

For numeric variables, we need to consider following:

-The user has to leave the default AGEP or specify their own selection of variables -The additional variables they specify must be valid -PWGTP should always be returned regardless of the user’s selection -There must be at least two numeric variables returned

To satisfy these conditions, we should:

  1. Make sure the user doesn’t pass NULL as their selection
  2. Make sure each variable specified is valid
  3. Add PWGTP to the vector of selected variables

Adding this workflow to our function, we have the following. I’m returning the vector of numeric variables to verify everything is working correctly:

query.census <- function(year = 2022, numeric_vars = "AGEP", categorical_vars = "SEX", 
                         geography = "State", specific_geo = "") {
  # check for valid year
  if(year == 2020 | !(year %in% 2010:2022)) stop("Error: Invalid year. Specify a year between 2010 and 2022 (not 2020)")
  
  # check for valid numeric vars
  numeric_vars_opts <- c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP")
  if(length(numeric_vars) == 0) stop("No numeric variable selected. Specify at least one.")
  for(i in numeric_vars) {
    if(!(i %in% numeric_vars_opts)) stop("Invalid numeric variable. Specify a single variable or a vector of variables. Options: AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP")
  }
  
  numeric_vars <- c(numeric_vars, "PWGTP")
  
  return(numeric_vars)
}

query.census <- function(year = 2022, numeric_vars = "AGEP", categorical_vars = "SEX", 
                         geography = "State", specific_geo = 3) {
  # check for valid year
  if(year == 2020 | !(year %in% 2010:2022)) stop("Error: Invalid year. Specify a year between 2010 and 2022 (not 2020)")
  
  # check for valid numeric vars
  numeric_vars_opts <- c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP")
  if(length(numeric_vars) == 0) stop("No numeric variable selected. Specify at least one.")
  for(i in numeric_vars) {
    if(!(i %in% numeric_vars_opts)) stop("Invalid numeric variable. Specify a single variable or a vector of variables. Options: AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP")
  }
  
  numeric_vars <- c(numeric_vars, "PWGTP")
  
  # check for valid categorical vars
  categorical_vars_opts <- c("FER", "HHL", "SCH", "SCHL", "SEX")
  if(length(categorical_vars) == 0) stop("No categorical variable selected. Specify at least one or leave SEX as the default")
  for(i in categorical_vars) {
    if(!(i %in% categorical_vars_opts)) stop("Invalid categorical variable. Specify a single variable or a vector of variables. Options: FER, HHL, SCH, SCHL, SEX")
  }
  
  # check for valid geography scope and region of interest
  geography_opts <- c("Region", "State", "Division")
  if(!(geography %in% geography_opts)) stop("Invalid geography classification. Specify one of: State, Region, Division")

    if(geography == "Division" & !(specific_geo %in% 1:9)) stop("Invalid division requested. Specify a number 1 to 9 or `all`")
  }
  
  return("API would be called")
}

query.census <- function(year = 2022, numeric_vars = "AGEP", categorical_vars = "SEX", 
                         geography = "State", specific_geo = 4) {
  # check for valid year
  if(year == 2020 | !(year %in% 2010:2022)) stop("Error: Invalid year. Specify a year between 2010 and 2022 (not 2020)")
  
  # check for valid numeric vars
  numeric_vars_opts <- c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP")
  if(length(numeric_vars) == 0) stop("No numeric variable selected. Specify at least one.")
  for(i in numeric_vars) {
    if(!(i %in% numeric_vars_opts)) stop("Invalid numeric variable. Specify a single variable or a vector of variables. Options: AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP")
  }
  
  numeric_vars <- c(numeric_vars, "PWGTP")
  
  # check for valid categorical vars
  categorical_vars_opts <- c("FER", "HHL", "SCH", "SCHL", "SEX")
  if(length(categorical_vars) == 0) stop("No categorical variable selected. Specify at least one or leave SEX as the default")
  for(i in categorical_vars) {
    if(!(i %in% categorical_vars_opts)) stop("Invalid categorical variable. Specify a single variable or a vector of variables. Options: FER, HHL, SCH, SCHL, SEX")
  }
  
  # check for valid geography scope and region of interest
  geography_opts <- c("Region", "State", "Division")
  if(!(geography %in% geography_opts)) stop("Invalid geography classification. Specify one of: State, Region, Division")
  if(specific_geo == "all") {
    specific_geo = "*"
  }
  else {
    if(geography == "State" & !(specific_geo %in% 1:50)) stop("Invalid state requested. Specify a number 1 to 50 or `all`")
    if(geography == "Region" & !(specific_geo %in% 1:4)) stop("Invalid region requested. Specify a number 1 to 4 or `all`")
    if(geography == "Division" & !(specific_geo %in% 1:9)) stop("Invalid division requested. Specify a number 1 to 9 or `all`")
  }
  
  api_key <- "59c3be738e0d051c63b57b6327de0b11ef0f7a1e"
  api_url <- paste0("https://api.census.gov/data/",
                    year,
                    "/acs/acs1/pums?get=",
                    paste0(numeric_vars, collapse = ","), # reformat numeric vars
                    ",",
                    paste0(categorical_vars, collapse = ","), # reformat categorical vars
                    "&for=",
                    tolower(geography), # region type
                    ":",
                    specific_geo, # specific region
                    "&key=",
                    api_key)
  
  api_return <- GET(api_url)
  api_parsed <- fromJSON(rawToChar(api_return$content)) |>
                as_tibble()
  
  # return just a preview to test func
  return(head(api_parsed))
}
